getCollaborationRequest
getCollaborationRequest(businessObjectType: string, businessObjectId: string, additionalFilters?: Array<FieldFilter>, forceRefresh: boolean)

Gets collaboration request from cache if available, otherwise fetches from API.

Example:

// For quotes - simple case
this.collaborationService.getCollaborationRequest('Proposal', quoteId).subscribe(request => ...);

// For orders with additional filters
this.collaborationService.getCollaborationRequest('Order', orderId, [
  { field: 'Status', value: 'Active', filterOperator: FilterOperator.EQUAL }
]).subscribe(request => ...);

// Force refresh from API
this.collaborationService.getCollaborationRequest('Proposal', quoteId, [], true).subscribe(request => ...);

retrieveCollaborationRequests
retrieveCollaborationRequests(filters: Array<FieldFilter>, sortField: string, sortDirection: "ASC" | "DESC", pageSize: number, pageNumber: number)

Retrieves collaboration requests from the API based on provided filters.

Example:

import { CollaborationRequestService, CollaborationRequest, FilterOperator } from '@congarevenuecloud/ecommerce';

export class MyComponent implements OnInit {
  collaborationRequests: CollaborationRequest[];

  constructor(private collaborationService: CollaborationRequestService) {}

  ngOnInit() {
    this.collaborationService.retrieveCollaborationRequests([
      { field: 'ParentBusinessObjectId', value: quoteId, filterOperator: FilterOperator.EQUAL }
    ]).subscribe(requests => this.collaborationRequests = requests);

    // Get collaboration requests with specific status
    this.collaborationService.retrieveCollaborationRequests([
      { field: 'ParentBusinessObjectType', value: 'Proposal', filterOperator: FilterOperator.EQUAL },
      { field: 'Status', value: 'Active', filterOperator: FilterOperator.EQUAL }
    ]).subscribe(requests => this.collaborationRequests = requests);

    // Get collaboration requests with custom pagination and sorting
    this.collaborationService.retrieveCollaborationRequests(
      [{ field: 'ParentBusinessObjectType', value: 'Proposal', filterOperator: FilterOperator.EQUAL }],
      'CreatedDate',
      'ASC',
      20,
      2
    ).subscribe(requests => this.collaborationRequests = requests);
  }
}
